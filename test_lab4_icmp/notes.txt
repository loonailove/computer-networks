struct icmp_hdr {   /* are exact 8 octeti  */
    uint8_t type;      // (0 = Reply, 8 = Request, 11 = Time Exceeded)
    uint8_t code;
    uint16_t checksum; // checksum for the entire ICMP header
    
    uint16_t id;
    uint16_t sequence;
} __attribute__((packed));

0 = Reply
8 = Request
11 = Time Exceeded
